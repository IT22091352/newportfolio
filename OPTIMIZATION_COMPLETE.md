💼 COMPONENT USAGE REFERENCE

In Hero.jsx:
  FROM: import AnimatedOrbs from './AnimatedOrbs';
  TO:   import AnimatedOrbsOptimized from './AnimatedOrbsOptimized';
  
  FROM: <AnimatedOrbs />
  TO:   <AnimatedOrbsOptimized />

  FROM: import TiltCard from './TiltCard';
  TO:   import TiltCardOptimized from './TiltCardOptimized';
  
  FROM: <TiltCard className="...">
  TO:   <TiltCardOptimized className="...">

In Any Component with Images:
  FROM: <img src={url} alt="description" />
  TO:   <OptimizedImage src={url} alt="description" loading="lazy" />

In App.js:
  ADD:  import { Suspense, lazy } from 'react';
  ADD:  const LazyProjects = lazy(() => import('./components/Projects'));
  ADD:  const LazyExperience = lazy(() => import('./components/Experience'));
  ADD:  const LazySkills = lazy(() => import('./components/Skills'));
  
  WRAP: <Suspense fallback={<LoadingFallback />}>
          <LazyProjects />
        </Suspense>

In Any Animated Component:
  ADD:  import { useIsMobile } from '../hooks/useMediaQuery';
  ADD:  const isMobile = useIsMobile();
  ADD:  animate={isMobile ? { opacity: 1 } : { /* complex anim */ }}
